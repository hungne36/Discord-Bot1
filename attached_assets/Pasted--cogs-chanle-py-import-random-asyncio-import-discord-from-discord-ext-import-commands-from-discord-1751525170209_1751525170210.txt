import random, asyncio

class ChanLeModal(discord.ui.Modal):
    def __init__(self, choice: str):
        title = f"C∆∞·ª£c {'Ch·∫µn' if choice=='chan' else 'L·∫ª'}"
        super().__init__(title=title)
        self.choice = choice
        self.amount = discord.ui.TextInput(label="S·ªë xu c∆∞·ª£c", placeholder="‚â•1.000", max_length=18)
        self.add_item(self.amount)

    async def on_submit(self, interaction: discord.Interaction):
        stake = int(self.amount.value)
        ok, wait = can_play(interaction.user.id)
        if not ok:
            return await interaction.response.send_message(f"‚è≥ ƒê·ª£i {int(wait)}s", ephemeral=True)
        bal = get_balance(interaction.user.id)
        if stake < 1000 or stake > bal:
            return await interaction.response.send_message("‚ùå C∆∞·ª£c kh√¥ng h·ª£p l·ªá!", ephemeral=True)

        await interaction.response.send_message("‚öñÔ∏è ƒêang ch·ªù...", ephemeral=True)
        await asyncio.sleep(2)
        s = random.randint(0,59)
        total = (s//10)+(s%10)
        kq = "chan" if total%2==0 else "le"
        win = (kq == self.choice)

        if win:
            profit = round(stake * 0.85)
            buff = get_pet_buff(interaction.user.id)
            bonus = round(profit * buff / 100) if buff else 0
            delta = stake + profit + bonus
        else:
            delta = -stake

        newb = update_balance(interaction.user.id, delta)
        add_history(interaction.user.id, f"chanle_{'win' if win else 'lose'}", delta, newb)

        txt = f"üî¢ S·ªë {s:02d} ‚Üí **{total}** ({'CH·∫¥N' if kq=='chan' else 'L·∫∫'})\n"
        txt += f"{'üéâ Th·∫Øng!' if win else 'üí∏ Thua!'} Œî={delta:+,}\nüí∞ D∆∞: {newb:,}"
        await interaction.followup.send(txt, ephemeral=False)
